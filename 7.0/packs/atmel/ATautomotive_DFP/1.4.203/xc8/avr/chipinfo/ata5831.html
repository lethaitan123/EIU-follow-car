// External Clock enable : CLEAR
// Disable external Reset : CLEAR
// Select interrupt vector location : CLEAR
// Preserve EEPROM memory through the Chip Erase cycle : CLEAR
// Watch-dog Timer always on : CLEAR
// Serial program downloading (SPI) enabled : SET
// Debug Wire enable : CLEAR
// Divide clock by 8 internally : CLEAR
#pragma config EXTCLKEN = CLEAR, RSTDISBL = CLEAR, BOOTRST = CLEAR, EESAVE = CLEAR, WDTON = CLEAR, SPIEN = SET, DWEN = CLEAR, CKDIV8 = CLEAR
